The first 3 are just messing around trying to find something I like that I could make iterations of.

void mainImage(out vec4 fragColor, in vec2 fragCoord) { vec2 uv = fragCoord / iResolution.xy; vec3 col = vec3( 0.5 + 0.5 * sin(uv.x * 10.0 + iTime), 0.5 + 0.5 * sin(uv.y * 10.0 + iTime), 0.5 + 0.5 * sin((uv.x + uv.y) * 10.0 + iTime) ); fragColor = vec4(col, 1.0); }

void mainImage(out vec4 fragColor, in vec2 fragCoord) { vec2 uv = fragCoord / iResolution.xy; uv -= 0.5; uv.x *= iResolution.x / iResolution.y; float angle = atan(uv.y, uv.x); float radius = length(uv); float tunnel = sin(10.0 * radius - iTime) * 0.5 + 0.5; vec3 col = vec3(tunnel, angle / 3.14159, radius); fragColor = vec4(col, 1.0); }

void mainImage(out vec4 fragColor, in vec2 fragCoord) { vec2 uv = fragCoord / iResolution.xy; vec3 col1 = vec3(uv.x, 1.0 - uv.y, sin(iTime)); vec3 col2 = vec3(cos(iTime), uv.y, 1.0 - uv.x); // Blend the colors vec3 col = mix(col1, col2, 0.5 + 0.5 * sin(iTime)); fragColor = vec4(col, 1.0); }

Messing around with more color variations to get a glow effect

void mainImage(out vec4 fragColor, in vec2 fragCoord) { vec2 uv = fragCoord / iResolution.xy; uv -= 0.5; float distance = length(uv); float ripple = sin(distance * 20.0 - iTime * 5.0); ripple += 0.5 * sin(uv.x * 10.0 + iTime); float glow = exp(-distance * 5.0); // Add a glow that fades with distance vec3 col = vec3( 1.0 * glow, (0.8 + 0.5 * ripple) * glow, (0.6 + 0.4 * abs(ripple)) * glow ); fragColor = vec4(col, 1.0); }

Went back to the normal ripples to make a kaleidascope effect

void mainImage(out vec4 fragColor, in vec2 fragCoord) { vec2 uv = fragCoord / iResolution.xy; uv -= 0.5; uv.x *= iResolution.x / iResolution.y; // Maintain aspect ratio float symmetry = 6.0; float angle = atan(uv.y, uv.x); angle = mod(angle, 3.14159 / symmetry) - (3.14159 / symmetry / 2.0); float distance = length(uv); float ripple = sin(distance * 20.0 - iTime * 5.0 + angle * symmetry); vec3 col = vec3( 1.0, 0.8 + 0.5 * ripple, 0.6 + 0.4 * abs(ripple) ); fragColor = vec4(col, 1.0); }
